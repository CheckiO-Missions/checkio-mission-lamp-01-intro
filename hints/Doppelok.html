Okay, okay! That's the spirit!=) Here's a one-line version using the zip() function. The logic is the same, only there are not two lists, but one with grouped pairs of light bulbs on and off. 
from datetime import datetime
from typing import List


def sum_light(els: List[datetime]) -> int:
    return sum([(b - a).total_seconds() for a, b in zip(els[::2], els[1::2])])
